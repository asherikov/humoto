123  /**
124  * @brief Determines position of support.
125  */
126  void determineSupportPosition()
127  {
128  switch(state_type_)
129  {
130  case STATE_LSS:
131  current_support_position_ = left_foot_.position_.head(2);
132  break;
133  case STATE_RSS:
134  current_support_position_ = right_foot_.position_.head(2);
135  break;
136  case STATE_DS:
137  /// Assume that in DS feet are aligned.
138  current_support_position_ = (right_foot_.position_.head(2) + left_foot_.position_.head(2)) / 2.;
139  break;
140  case STATE_TDS:
141  switch(next_state_type_)
142  {
143  case STATE_LSS:
144  current_support_position_ = left_foot_.position_.head(2);
145  break;
146  case STATE_RSS:
147  current_support_position_ = right_foot_.position_.head(2);
148  break;
149  case STATE_DS:
150  /// Assume that in DS feet are aligned.
151  current_support_position_ = (right_foot_.position_.head(2) + left_foot_.position_.head(2)) / 2.;
152  break;
