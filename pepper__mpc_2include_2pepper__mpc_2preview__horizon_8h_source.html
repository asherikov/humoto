136  /**
137  * @brief Preview horizon of an MPC [form_preview_horizon.m]
138  */
139  class HUMOTO_LOCAL PreviewHorizon
140  {
141  public:
142  std::vector<PreviewHorizonInterval> intervals_;
143 
144 
145  public:
146  /**
147  * @brief Form the preview horizon object
148  *
149  * @param[in] mpc_params
150  * @param[in] model
151  * @param[in] motion_parameters
152  *
153  * @return true if successful
154  */
155  bool form( const MPCParameters &mpc_params,
156  const Model &model,
157  const MotionParameters &motion_parameters)
158  {
159  bool preview_horizon_formed = false;
160 
161 
162  if ((motion_parameters.duration_ms_ == MotionParameters::UNLIMITED_DURATION)
163  ||
164  ( static_cast<std::ptrdiff_t>(mpc_params.preview_horizon_length_ * mpc_params.sampling_time_ms_)
165  <= motion_parameters.duration_ms_ ))
166  {
167  double theta = model.getBaseOrientation();
168 
169  intervals_.resize(mpc_params.preview_horizon_length_);
170 
171  for (std::size_t i = 0; i < mpc_params.preview_horizon_length_; ++i)
172  {
173  intervals_[i].initialize(theta, mpc_params, motion_parameters, model);
174  }
175 
176  preview_horizon_formed = true;
177  }
178 
179  return (preview_horizon_formed);
180  }
181 
182 
183  /**
184  * @brief Form the preview horizon object
185  *
186  * @param[in] mpc_params
187  * @param[in] model
188  * @param[in] motion_parameters_deque
189  *
190  * @return true if successful
191  */
192  bool form( const MPCParameters &mpc_params,
193  const Model &model,
194  const std::deque<MotionParameters> &motion_parameters_deque)
195  {
196  bool preview_horizon_formed = false;
197 
198 
199  double theta = model.getBaseOrientation();
200  intervals_.resize(mpc_params.preview_horizon_length_);
201 
202 
203  std::size_t i = 0;
204  std::size_t remainder_ms = 0;
205  for ( std::deque<MotionParameters>::const_iterator it = motion_parameters_deque.begin();
206  it != motion_parameters_deque.end();
207  ++it)
208  {
209  if (it->duration_ms_ == MotionParameters::UNLIMITED_DURATION)
210  {
211  for (; i < mpc_params.preview_horizon_length_; ++i)
212  {
213  intervals_[i].initialize(theta, mpc_params, *it, model);
214  }
215  preview_horizon_formed = true;
216  break;
217  }
218  else
219  {
220  std::size_t number_of_iterations = std::min(
221  static_cast<std::size_t>( (remainder_ms + it->duration_ms_) / mpc_params.sampling_time_ms_),
222  mpc_params.preview_horizon_length_ - i);
223 
224  for (std::size_t j = 0; j < number_of_iterations; ++i, ++j)
225  {
226  intervals_[i].initialize(theta, mpc_params, *it, model);
227  }
228 
229  if (i == mpc_params.preview_horizon_length_)
230  {
231  preview_horizon_formed = true;
232  break;
233  }
234  else
235  {
236  remainder_ms = static_cast<std::size_t>(
237  (remainder_ms + it->duration_ms_) % mpc_params.sampling_time_ms_);
238  }
239  }
240  }
241 
242  return (preview_horizon_formed);
243  }
