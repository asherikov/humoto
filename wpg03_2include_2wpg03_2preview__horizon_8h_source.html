97  /**
98  * @brief Form the preview horizon object
99  *
100  * @param[in] model
101  * @param[in] mpc_params
102  * @param[in] walk_fsm
103  *
104  */
105  bool PreviewHorizon::form( const MPCParameters &mpc_params,
106  const Model &model,
107  const WalkFiniteStateMachine &walk_fsm)
108  {
109  bool preview_horizon_formed = true;
110 
111 
112  // Length of the preview horizon (N)
113  unsigned int preview_horizon_length_ = mpc_params.preview_horizon_length_;
114 
115  unsigned int walk_index = walk_fsm.index_;
116 
117  unsigned int num_intervals_left = 0;
118 
119 
120  //is_new_sample_ = walk_fsm.is_new_sample_;
121  states_.clear();
122  variable_steps_indices_.clear();
123 
124  std::vector<unsigned int> support_indices;
125  unsigned int pwin_state_index = 0;
126 
127  intervals_.resize(preview_horizon_length_);
128 
129  unsigned int interval_index = 0;
130  while(interval_index < preview_horizon_length_)
131  {
132  if(walk_index >= walk_fsm.states_.size())
133  {
134  preview_horizon_formed = false;
135  return (preview_horizon_formed);
136  }
137 
138  WalkState state = walk_fsm.states_.at(walk_index);
139 
140  PreviewHorizonInterval interval;
141  interval.state_index_ = pwin_state_index; // in TDS points to the next SS
142  interval.cvel_ref_ = state.R_cvel_ref_;
143  interval.hg_ = model.getComHeight()/ humoto::g_gravitational_acceleration; // Height of the CoM / gravitational acceleration
144  interval.omega_ = std::sqrt(1./interval.hg_);
145 
146  support_indices.push_back(interval_index);
147 
148  if (pwin_state_index == 0)
149  {
150  // current sampling interval (the 1st in the preview window)
151  if (state.type_ == STATE_TDS)
152  {
153  interval.T_ = convertMillisecondToSecond(walk_fsm.current_step_duration_ms_);
154  ++walk_index;
155  state = walk_fsm.states_.at(walk_index);
156  num_intervals_left = floor(state.duration_ms_ / mpc_params.sampling_time_ms_);
157  }
158  else
159  {
160  interval.T_ = convertMillisecondToSecond(walk_fsm.current_step_duration_ms_ % mpc_params.sampling_time_ms_);
161  num_intervals_left = floor(walk_fsm.current_step_duration_ms_ / mpc_params.sampling_time_ms_);
162 
163  if ((walk_fsm.current_step_duration_ms_ % mpc_params.sampling_time_ms_) == 0)
164  {
165  interval.T_ = mpc_params.getSamplingTime();
166  --num_intervals_left;
167  }
168  }
169 
170  intervals_[interval_index] = interval;
171  ++interval_index;
172  }
173  else
174  {
175  if (state.type_ != STATE_TDS)
176  {
177  HUMOTO_THROW_MSG("Unexpected state type!");
178  }
179  // new variable footstep
180  variable_steps_indices_.push_back(interval_index);
181 
182  // transitional DS
183  interval.T_ = mpc_params.getTDSSamplingTime();
184 
185 
186  intervals_[interval_index] = interval;
187  ++interval_index;
188 
189  // next SS
190  ++walk_index;
191  state = walk_fsm.states_.at(walk_index);
192 
193  num_intervals_left = floor(state.duration_ms_ / mpc_params.sampling_time_ms_);
194  }
195 
196  interval.T_ = mpc_params.getSamplingTime();
197 
198  // fill the other intervals
199  unsigned int max_index = std::min(preview_horizon_length_, interval_index + num_intervals_left);
200  for(unsigned int i = interval_index; i < max_index; ++i)
201  {
202  intervals_[i] = interval;
203  }
204 
205  interval_index += num_intervals_left;
206  get_constraints(model, walk_fsm, walk_index, state.fd_bounds_, state.cop_bounds_);
207  states_.push_back(state);
208  ++pwin_state_index;
209  ++walk_index;
210  }
211 
212  return (preview_horizon_formed);
213  }

216  /**
217  * @brief Utility Function
218  *
219  * @param[in] model
220  * @param[in] walk_fsm
221  * @param[in] walk_index
222  * @param[out] fd_bounds
223  * @param[out] cop_bounds
224  */
225  void PreviewHorizon::get_constraints(const Model &model,
226  const WalkFiniteStateMachine &walk_fsm,
227  unsigned int walk_index,
228  etools::Matrix2 &fd_bounds,
229  etools::Matrix2 &cop_bounds)
230  {
231  ModelStateType current_state_type = walk_fsm.states_.at(walk_index).type_;
232  if (walk_index == 0)
233  {
234  fd_bounds << 0., 0., 0., 0.;
235  }
236  else
237  {
238  ModelStateType preceding_state_type = walk_fsm.states_.at(walk_index-2).type_;
239  switch(current_state_type)
240  {
241  case STATE_LSS:
242  switch(preceding_state_type)
243  {
244  case STATE_DS:
245  fd_bounds = model.getFootBounds(humoto::LeftOrRight::LEFT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_ADS);
246  break;
247  case STATE_RSS:
248  fd_bounds = model.getFootBounds(humoto::LeftOrRight::LEFT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_SS);
249  break;
250  default:
251  HUMOTO_THROW_MSG("unexpected support type");
252  break;
253  }
254  break;
255 
256  case STATE_RSS:
257  switch(preceding_state_type)
258  {
259  case STATE_DS:
260  fd_bounds = model.getFootBounds(humoto::LeftOrRight::RIGHT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_ADS);
261  break;
262  case STATE_LSS:
263  fd_bounds = model.getFootBounds(humoto::LeftOrRight::RIGHT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_SS);
264  break;
265  default:
266  HUMOTO_THROW_MSG("unexpected support type");
267  break;
268  }
269  break;
270 
271  case STATE_DS:
272  switch(preceding_state_type)
273  {
274  case STATE_LSS:
275  fd_bounds = model.getFootBounds(humoto::LeftOrRight::RIGHT, humoto::walking::FootBoundsType::ALIGNED_TO_SS);
276  break;
277  case STATE_RSS:
278  fd_bounds = model.getFootBounds(humoto::LeftOrRight::LEFT, humoto::walking::FootBoundsType::ALIGNED_TO_SS);
279  break;
280  default:
281  HUMOTO_THROW_MSG("unexpected support type");
282  break;
283  }
284  break;
285 
286  default:
287  HUMOTO_THROW_MSG("unexpected support type");
288  break;
289  }
290  }
291 
292 
293  if (current_state_type == STATE_DS)
294  {
295  cop_bounds = model.getADSCoPBounds();
296  }
297  else
298  {
299  cop_bounds = model.getSSCoPBounds();
300  }
301  }
