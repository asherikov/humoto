474  HUMOTO_ASSERT( true == parameters_.solve_two_levels_as_qp_,
475  "Interpretation of hierarchies as QPs is disabled.");
476 
477  std::size_t number_of_levels = hierarchy.getNumberOfLevels();

482

484 
485  HUMOTO_ASSERT( hierarchy[objective_level].isEquality(),
486  "This type of hierarchy cannot be solved with a QP solver.");
487 
488  HUMOTO_ASSERT( hierarchy[objective_level].getNumberOfConstraints() > 0,
489  "Empty objective.");
