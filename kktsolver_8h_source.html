171  /**
172  * @brief Eliminate constraints and solve (only diagonal Hessians are supported).
173  *
174  * @param[out] solution
175  * @param[in] hierarchy
176  */
177  void solveWithElimination( humoto::Solution &solution,
178  const humoto::OptimizationProblem &hierarchy)
179  {
180  std::ptrdiff_t num_var = solution.getNumberOfVariables();
181 
182  Eigen::VectorXd inverted_H;
183  inverted_H.resize(num_var);
184  for (EigenIndex i = 0; i < inverted_H.size(); ++i)
185  {
186  inverted_H(i) = 1./(H_(i,i) + parameters_.elimination_regularization_factor_);
187  }
188 
189  Eigen::VectorXd iH_g = inverted_H.asDiagonal() * g_;
190  Eigen::MatrixXd iH_At = inverted_H.asDiagonal() * equality_constraints_.getA().transpose();
191 
192  solution.x_.noalias() = iH_At
193  *
194  // lambda
195  (equality_constraints_.getA() * iH_At).llt().solve(
196  equality_constraints_.getA()*iH_g + equality_constraints_.getB())
197  -
198  iH_g;
199 
200  solution.return_status_ = SolverStatus::OK;
201  }

204  /**
205  * @brief Solve KKT system directly
206  *
207  * @param[out] solution
208  * @param[in] hierarchy
209  */
210  void solveDirect( humoto::Solution &solution,
211  const humoto::OptimizationProblem &hierarchy)
212  {
213  Eigen::MatrixXd kkt_matrix;
214  Eigen::VectorXd kkt_vector;
215 
216  std::ptrdiff_t num_var = solution.getNumberOfVariables();
217  std::ptrdiff_t num_ctr = equality_constraints_.getNumberOfConstraints();
218 
219  kkt_matrix.resize(num_var+num_ctr, num_var+num_ctr);
220  kkt_vector.resize(num_var+num_ctr);
221 
222  kkt_matrix << H_, equality_constraints_.getA().transpose(),
223  equality_constraints_.getA(), Eigen::MatrixXd::Zero(num_ctr, num_ctr);
224 
225  kkt_vector << -g_,
226  equality_constraints_.getB();
227 
228 
229  Eigen::VectorXd x;
230  switch(parameters_.solution_method_)
231  {
232  case SolverParameters::LU:
233  x = kkt_matrix.lu().solve(kkt_vector);
234  break;
235  case SolverParameters::QR:
236  x = kkt_matrix.colPivHouseholderQr().solve(kkt_vector);
237  break;
238  default:
239  HUMOTO_THROW_MSG("Unknown solution method.");
240  }
241 
242  solution.return_status_ = SolverStatus::OK;
243 
244  solution.x_ = x.segment(0, num_var);
245  }
