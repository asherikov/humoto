108  /**
109  * @brief Condense output of the system.
110  *
111  * output = Ox*x0 + Ou*(u0,...,uN)
112  *
113  * @tparam t_num_vars number of state variables
114  * @tparam t_num_outputs number of output variables
115  * @tparam t_DMatrix D matrix, should be an Eigen matrix or a scalar
116  * @tparam t_EMatrix E matrix, should be an Eigen matrix or a scalar
117  *
118  * @param[out] Ox matrix Ox
119  * @param[out] Ou vector of Ou matrices
120  * @param[in] D vector of D matrices
121  * @param[in] E vector of E matrices
122  * @param[in] Ux matrix Ux
123  * @param[in] Uu vector of Uu matrices
124  */
125  template<int t_num_vars,
126  int t_num_outputs,
127  typename t_DMatrix,
128  typename t_EMatrix>
129  void condenseOutput(etools::GenericBlockMatrix<t_num_outputs, t_num_vars> &Ox,
130  std::vector< etools::LeftLowerTriangularBlockMatrix<t_num_outputs,
131  etools::MatrixBlockSizeType::DYNAMIC > > &Ou,
132  const std::vector<t_DMatrix> &D,
133  const std::vector<t_EMatrix> &E,
134  const etools::GenericBlockMatrix<t_num_vars, t_num_vars> &Ux,
135  const std::vector< etools::LeftLowerTriangularBlockMatrix< t_num_vars,
136  etools::MatrixBlockSizeType::DYNAMIC > > &Uu)
137  {
138  std::ptrdiff_t N = Ux.getNumberOfBlocksVertical();
139 
140  HUMOTO_ASSERT(Ux.getNumberOfBlocksHorizontal() == 1,
141  "Wrong size of Ux matrix.")
142  HUMOTO_ASSERT(D.size() == E.size(), "Mismatching number of D and E matrices.")
143  HUMOTO_ASSERT(E.size() == static_cast<std::size_t>(N), "Wrong number of matrices.")
144 
145 
146  std::ptrdiff_t num_controls = 0;
147  std::vector<std::ptrdiff_t> control_offsets;
148 
149 
150  Ox.resize(N, 1);
151  Ox(0, 0) = D[0];
152 
153  Ou.resize(Uu.size());
154  control_offsets.resize(Uu.size());
155  for (std::size_t i = 0; i < Uu.size(); ++i)
156  {
157  control_offsets[i] = num_controls;
158  num_controls += Uu[i].getBlockColsNum();
159  Ou[i].setBlockSize(etools::MatrixBlockSizeType::UNDEFINED, Uu[i].getBlockColsNum());
160  Ou[i].setZero(N, N);
161  }
162 
163  for (std::size_t i = 0; i < Uu.size(); ++i)
164  {
165  Ou[i](N-1, N-1) = E[0].block( 0,
166  control_offsets[i],
167  t_num_outputs,
168  Uu[i].getBlockColsNum());
169  }
170 
171  for (std::ptrdiff_t i = 1; i < N; ++i)
172  {
173  Ox(i, 0).noalias() = D[i] * Ux(i-1, 0);
174 
175  for (std::size_t j = 0; j < Uu.size(); ++j)
176  {
177  Ou[j].row(i, 0, i).noalias() = D[i] * Uu[j].row(i-1, 0, i);
178  Ou[j](i, i) = E[i].block( 0,
179  control_offsets[j],
180  t_num_outputs,
181  Uu[j].getBlockColsNum());
182  }
183  }
184  }

246  * @brief Create the condensed matrices (S,U) of a Model Predictive Control problem such that X = S*x0 + U*u
247  *
248  * @tparam t_num_vars number of variables
249  * @tparam t_num_controls number of controls
250  * @tparam t_AMatrix A matrix, should be an Eigen matrix or a scalar
251  * @tparam t_BMatrix B matrix, should be an Eigen matrix
252  *
253  * @param[out] S
254  * @param[out] U
255  * @param[in] A
256  * @param[in] B
257  */
258  template<int t_num_vars,
259  int t_num_controls,
260  typename t_AMatrix,
261  typename t_BMatrix,
262  etools::MatrixSparsityType::Type t_bmatrix_sparsity_type>
263  void condense( etools::GenericBlockMatrix<t_num_vars, t_num_vars> &S,
264  etools::BlockMatrix<t_num_vars, t_num_controls, t_bmatrix_sparsity_type> &U,
265  const std::vector<t_AMatrix> &A,
266  const std::vector<t_BMatrix> &B)
267  {
268  HUMOTO_ASSERT(A.size() == B.size(), "Mismatching number of A and B matrices.")
269  HUMOTO_ASSERT(A.size() > 0, "Wrong number of matrices.")
270 
271  std::ptrdiff_t N = A.size();
272 
273  S.resize(N, 1);
274  U.setZero(N, N);
275 
276  // form U (right to left)
277  // S is partially formed in the process
278  S(N-1, 0).setIdentity();
279  // last column
280  U(N-1, N-1) = B[N-1];
281  // other columns
282  for(std::ptrdiff_t k=N-2; k>=0; --k)
283  {
284  //S.column(0, k+1, N-k-1) *= A[k+1];
285  //loop is faster for blocks with constant size
286  for (std::ptrdiff_t i = k+1; i < N; ++i)
287  {
288  S(i, 0) *= A[k+1];
289  }
290  S(k, 0).setIdentity();
291 
292  U(k, k) = B[k];
293  //U.column(k, k+1, N-k-1).noalias() = S.column(0, k+1, N-k-1) * B[k];
294  //loop is faster for blocks with constant size
295  for (std::ptrdiff_t i = k+1; i < N; ++i)
296  {
297  U(i, k).noalias() = S(i, 0) * B[k];
298  }
299  }
300 
301 
302  // finalize formulation of S
303  S.column(0) *= A[0];
304  }

307  /**
308  * @brief Create the condensed matrices (S,U) of a Model Predictive Control problem such that X = S*x0 + U*u
309  *
310  * @tparam t_AMatrix A matrix, should be an Eigen matrix or a scalar
311  * @tparam t_BMatrix B matrix, should be an Eigen matrix
312  *
313  * @param[out] S
314  * @param[out] U
315  * @param[in] A
316  * @param[in] B
317  */
318  template<typename t_AMatrix, typename t_BMatrix>
319  void condense( Eigen::MatrixXd &S,
320  Eigen::MatrixXd &U,
321  const std::vector<t_AMatrix> &A,
322  const std::vector<t_BMatrix> &B)
323  {
324  HUMOTO_ASSERT(A.size() == B.size(), "Mismatching number of A and B matrices.")
325  HUMOTO_ASSERT(A.size() > 0, "Wrong number of matrices.")
326  HUMOTO_ASSERT((A[0].rows() == B[0].rows()) && (A[0].cols() == B[0].rows()), "Wrong size of matrices.");
327 
328  int N = A.size();
329  int Ns = B[0].rows();
330  int Nu = B[0].cols();
331 
332 
333  S.resize(Ns * N, Ns);
334  U.resize(Ns*N, Nu*N);
335 
336  // form U (right to left)
337  // S is partially formed in the process
338  U.setZero();
339  S.block((N-1)*Ns, 0, Ns, Ns).setIdentity();
340  // last column
341  U.block((N-1)*Ns, (N-1)*Nu, Ns, Nu) = B[N-1];
342  // other columns
343  for(int k=N-2; k>=0; --k)
344  {
345  S.block((k+1)*Ns, 0, (N-k-1)*Ns, Ns) *= A[k+1];
346  S.block(k*Ns, 0, Ns, Ns).setIdentity();
347 
348  U.block((k)*Ns, k*Nu, Ns, Nu) = B[k];
349  U.block((k+1)*Ns, k*Nu, (N-k-1)*Ns, Nu).noalias() = S.block((k+1)*Ns, 0, (N-k-1)*Ns, Ns) * B[k];
350  }
351 
352 
353  // finalize formulation of S
354  S *= A[0];
355  }

358  /**
359  * @brief Create the condensed matrices (S,U) of a Time Invariant (constant A,B)
360  * Model Predictive Control problem such that X = S*x0 + U*u
361  *
362  * @param[out] S
363  * @param[out] U
364  * @param[in] preview_horizon_len length of the preview horizon
365  * @param[in] A
366  * @param[in] B
367  */
368  void condenseTimeInvariant( Eigen::MatrixXd &S,
369  Eigen::MatrixXd &U,
370  const std::size_t preview_horizon_len,
371  const Eigen::MatrixXd &A,
372  const Eigen::MatrixXd &B)
373  {
374  // form S
375  Eigen::MatrixXd M = A;
376  int num_rows_A = A.rows();
377  int num_cols_A = A.cols();
378  S.resize(num_rows_A*preview_horizon_len, num_cols_A);
379  S.block(0, 0 , num_rows_A, num_cols_A) = M;
380 
381  for(std::size_t i=1; i<preview_horizon_len; ++i)
382  {
383  M = M * A;
384  S.block(i*num_rows_A, 0 , num_rows_A, num_cols_A) = M;
385  }
386 
387  // form U
388  int num_rows_B = B.rows();
389  int num_cols_B = B.cols();
390  Eigen::MatrixXd Ucol(num_rows_B*preview_horizon_len, num_cols_B);
391  Ucol << B,
392  S.block(0, 0 , num_rows_B*(preview_horizon_len-1), num_cols_A) * B;
393  U.resize(num_rows_B*preview_horizon_len, num_cols_B*preview_horizon_len);
394  U.setZero();
395  for(std::size_t i=0; i<preview_horizon_len; ++i)
396  {
397  U.block(num_rows_B*i, num_cols_B*i, num_rows_B*(preview_horizon_len-i), num_cols_B) = Ucol.block(0, 0, num_rows_B*(preview_horizon_len-i), num_cols_B);
398  }
399  }
