41  * @brief Create the rotation matrices of the MPC problem
42  */
43  void formRotationMatrices()
44  {
45  std::vector<Eigen::MatrixXd> pwin_rotation_matrices;
46 
47  for(unsigned int i=0; i < (preview_horizon_.states_.size() - 1); ++i)
48  {
49  pwin_rotation_matrices.push_back(preview_horizon_.states_.at(i).rotation_);
50  }
51  R_ = etools::makeBlockDiagonal(pwin_rotation_matrices);
52 
53  pwin_rotation_matrices.clear();

56  pwin_rotation_matrices.push_back( preview_horizon_.states_.at(
57  preview_horizon_.intervals_.at(i).
58  state_index_).rotation_);

60  Rh_ = etools::makeBlockDiagonal(pwin_rotation_matrices);

64  /**
65  * @brief Create the foot position matrices
66  *
67  * @param[in] model
68  */
69  void formFootPosMatrices(const humoto::wpg03::Model &model)
70  {
71  // generate_fixed_step_vector
72  Eigen::MatrixXd Vc = etools::Matrix2::Identity().replicate(preview_horizon_.intervals_.size(), 1);
73  V0c_ = Vc * model.current_support_position_;
74 
75  // generate_var_step_matrix
76  V_ = Eigen::MatrixXd::Zero( preview_horizon_.intervals_.size()*2,
77  preview_horizon_.getNumberOfVariableSteps()*2);
78 
79  for(unsigned int i=0; i < preview_horizon_.variable_steps_indices_.size(); ++i)
80  {
81  unsigned int row_size = preview_horizon_.intervals_.size() - preview_horizon_.variable_steps_indices_[i];
82  V_.block(preview_horizon_.variable_steps_indices_[i]*2 , i*2, row_size*2, 2) =
83  etools::Matrix2::Identity().replicate(row_size, 1);
84  }
85 
86  V_ = V_*R_;
87 
88  Vfp_ = etools::Matrix2::Identity().replicate(
89  preview_horizon_.getNumberOfVariableSteps(),
90  preview_horizon_.getNumberOfVariableSteps()).triangularView<Eigen::Lower>() * R_;
91 
92  vfp_ = model.current_support_position_.replicate(preview_horizon_.getNumberOfVariableSteps(), 1);
93  }

96  /**
97  * @brief Process solution
98  *
99  * @param[in] solution solution
100  * @param[in] mpc_type
101  */
102  void processSolution(const humoto::Solution &solution,
103  const humoto::wpg03::MPCInteractionType mpc_type = MPC_NORMAL)
104  {
105  Eigen::VectorXd mpc_variables;
106  Eigen::VectorXd cop_local;
107  Eigen::VectorXd footpos_local;
108 
109  comjerk_profile = solution.getSolutionPart(COMJERK_VARIABLES_ID);
110  footpos_local = solution.getSolutionPart(FOOTPOS_VARIABLES_ID);
111 
112  switch(mpc_type)
113  {
114  case MPC_NORMAL:
115  case MPC_FOLLOWER:
116  {
117  mpc_variables.resize(comjerk_profile.rows() + footpos_local.rows());
118  mpc_variables << comjerk_profile, footpos_local;
119  break;
120  }
121  case MPC_LEADER:
122  {
123  extwrench_profile = solution.getSolutionPart(EXTWRENCH_VARIABLES_ID);
124  mpc_variables.resize(comjerk_profile.rows() + footpos_local.rows() + extwrench_profile.rows());
125  mpc_variables << comjerk_profile, footpos_local, extwrench_profile;
126  break;
127  }
128  default:
129  {
130  HUMOTO_THROW_MSG("unexpected MPC type");
131  break;
132  }
133  }
134 
135  cop_local = Sz_* mpc_variables + sz_;
136 
137  cop_profile = Rh_ * cop_local + V_ * footpos_local + V0c_;
138  cstate_profile = S_ * mpc_variables + s_;
139 
140  footpos_profile = Vfp_ * footpos_local + vfp_;
141  }

207  /**
208  * @brief Update control problem
209  *
210  * @param[in] model model of the system
211  * @param[in] walk_fsm walking finite state machine
212  * @param[in] mpc_type interaction behavior type
213  * @param[in] external_wrenches 6N vector of future forces and torques acting on the Center of Mass frame for N preview steps
214  *
215  * @return ControlProblemStatus::OK/ControlProblemStatus::STOPPED
216  */
217  ControlProblemStatus::Status
218  update( const humoto::wpg03::Model &model,
219  const humoto::wpg03::WalkFiniteStateMachine &walk_fsm,
220  const Eigen::VectorXd &external_wrenches,
221  const humoto::wpg03::MPCInteractionType mpc_type = MPC_NORMAL)
222  {
223  humoto::ControlProblemStatus::Status control_status = ControlProblemStatus::OK;
224 
225 
226  if (preview_horizon_.form( mpc_parameters_,
227  model,
228  walk_fsm) )
229  {
230  // decision variables
231  sol_structure_.reset();
232  sol_structure_.addSolutionPart(COMJERK_VARIABLES_ID, mpc_parameters_.preview_horizon_length_ * 2);
233  sol_structure_.addSolutionPart(FOOTPOS_VARIABLES_ID, preview_horizon_.getNumberOfVariableSteps() * 2);
234  switch(mpc_type)
235  {
236  case MPC_NORMAL:
237  case MPC_FOLLOWER:
238  {
239  break;
240  }
241  case MPC_LEADER:
242  {
243  sol_structure_.addSolutionPart(EXTWRENCH_VARIABLES_ID, mpc_parameters_.preview_horizon_length_ * 4);
244  break;
245  }
246  default:
247  {
248  HUMOTO_THROW_MSG("unexpected MPC type");
249  break;
250  }
251  }
252 
253 
254  unsigned int number_of_state_variables = model.Ns_ * mpc_parameters_.preview_horizon_length_;
255 
256 
257  // Initialize matrices
258  formRotationMatrices();
259  formFootPosMatrices(model);
260 
261 
262  // condensing
263  std::vector<Eigen::MatrixXd> A_matrices;
264  std::vector<Eigen::MatrixXd> B_matrices;
265  std::vector<Eigen::MatrixXd> D_matrices;
266  std::vector<Eigen::MatrixXd> F_matrices;
267 
268  Eigen::VectorXd cvel_ref;
269 
270  A_matrices.resize(mpc_parameters_.preview_horizon_length_);
271  B_matrices.resize(mpc_parameters_.preview_horizon_length_);
272  D_matrices.resize(mpc_parameters_.preview_horizon_length_);
273  F_matrices.resize(mpc_parameters_.preview_horizon_length_);
274 
275  cvel_ref.resize(mpc_parameters_.preview_horizon_length_*2);
276 
277  Eigen::VectorXd f_pred;
278  switch(mpc_type)
279  {
280  case MPC_NORMAL:
281  {
282  // process the external wrenches to get the ordering of needed terms
283  f_pred = getReorderedSet(external_wrenches);
284  break;
285  }
286  case MPC_FOLLOWER:
287  {
288  // process the external wrenches to get the ordering of needed terms
289  f_pred = getReorderedSet(external_wrenches);
290  Select_Fxy_ = etools::makeBlockDiagonal(selectxy_mat, mpc_parameters_.preview_horizon_length_) * f_pred;
291  break;
292  }
293  case MPC_LEADER:
294  {
295  break;
296  }
297  default:
298  {
299  HUMOTO_THROW_MSG("unexpected MPC type");
300  break;
301  }
302  }
303  for (unsigned int i = 0; i < mpc_parameters_.preview_horizon_length_; ++i)
304  {
305  double T = preview_horizon_.intervals_[i].T_;
306  double omega = preview_horizon_.intervals_[i].omega_;
307  double f_ext_z = external_wrenches(i*6+2);
308 
309  A_matrices[i] = model.getA6(T);
310  B_matrices[i] = model.getB6(T);
311 
312  // Note: in the math, these matrices start at preview step 1 and end at N
313  // it assumes that the given external_wrenches are the same with other variables constant over N
314  D_matrices[i] = model.getD6(omega, model.getMass(), f_ext_z);
315  F_matrices[i] = model.getF4(model.getComHeight(), model.getMass(), f_ext_z);
316 
317  cvel_ref.segment(i*2, 2) = preview_horizon_.intervals_[i].cvel_ref_;
318  }
319 
320  Eigen::MatrixXd Ux;
321  Eigen::MatrixXd Uu;
322 
323  condense(Ux, Uu, A_matrices, B_matrices);
324 
325  S_.resize (number_of_state_variables, sol_structure_.getNumberOfVariables ());
326  S_.setZero();
327  S_.topLeftCorner(Uu.rows(), Uu.cols()) = Uu;
328 
329 
330  s_ = Ux*model.cstate_;
331 
332  Sv_ = etools::selectRows(S_, 3, 1);
333  sv_ = etools::selectRows(s_, 3, 1) - cvel_ref;
334 
335  Eigen::MatrixXd Ox;
336  Ox.resize(2*mpc_parameters_.preview_horizon_length_, model.Ns_);
337  Ox = etools::makeBlockDiagonal(D_matrices) * Ux;
338 
339  Eigen::MatrixXd Ou;
340  Ou.setZero(2*mpc_parameters_.preview_horizon_length_, model.Nu_*mpc_parameters_.preview_horizon_length_);
341  Ou = etools::makeBlockDiagonal(D_matrices) * Uu;
342 
343  Sz_.resize(2*mpc_parameters_.preview_horizon_length_, sol_structure_.getNumberOfVariables ());
344 
345  switch(mpc_type)
346  {
347  case MPC_NORMAL:
348  case MPC_FOLLOWER:
349  {
350  F_ = etools::makeBlockDiagonal(F_matrices) * f_pred;
351 
352  Sz_ << Ou, -V_;
353  sz_ = Ox*model.cstate_ + F_ - V0c_;
354 
355  break;
356  }
357  case MPC_LEADER:
358  {
359  Eigen::MatrixXd Of; // TODO: refactor to better mimic math notation
360  Of = etools::makeBlockDiagonal(F_matrices);
361 
362  Sz_ << Ou, -V_, Of;
363  sz_ = Ox*model.cstate_ - V0c_;
364  break;
365  }
366  default:
367  {
368  HUMOTO_THROW_MSG("unexpected MPC type");
369  break;
370  }
371  }
372  Sz_ = Rh_.transpose()*Sz_;
373  sz_ = Rh_.transpose()*sz_;
374 
375  control_status = ControlProblemStatus::OK;
376  }
377  else
378  {
379  control_status = ControlProblemStatus::STOPPED;
380  }
381 
382  return (control_status);
383  }

388  *
389  * @param[in] solution solution

396  humoto::wpg03::ModelState getNextModelState(

444  /**
445  * @brief Set tracking gains individually
446  *
447  * @param[in] gain_x_m feed-forward acceleration gain (inertia) in x
448  * @param[in] gain_x_b velocity gain (damping) in x
449  * @param[in] gain_x_k position gain (stiffness) in x
450  * @param[in] gain_y_m feed-forward acceleration gain (inertia) in y
451  * @param[in] gain_y_b velocity gain (damping) in y
452  * @param[in] gain_y_k position gain (stiffness) in y
453  */
454  void setTrackingGains(const double gain_x_m, const double gain_x_b, const double gain_x_k,
455  const double gain_y_m, const double gain_y_b, const double gain_y_k)
456  {
457  Eigen::Matrix<double, 2, 6> gains;
458  // This rescales the impedance gains by normalizing on the velocity
459  // It allows the weighting of task-comimpedance to be similar to task-comvelocity
460  // and handles zero damping cases (not usual) without rescaling
461  if((gain_x_b == 0.) && (gain_y_b == 0.))
462  {
463  gains << gain_x_k, 0., gain_x_m, 0., 0., 0.,
464  0., 0., 0., gain_y_k, 0., gain_y_m;
465  selectxy_mat << 0., 1., 0., 0,
466  0., 0., 0., 1.;
467  }
468  else if((gain_x_b == 0.) && (gain_y_b != 0.))
469  {
470  gains << gain_x_k, 0., gain_x_m, 0., 0., 0.,
471  0., 0., 0., gain_y_k/gain_y_b, 1., gain_y_m/gain_y_b;
472  selectxy_mat << 0., 1., 0., 0,
473  0., 0., 0., 1./gain_y_b;
474  }
475  else if((gain_x_b != 0.) && (gain_y_b == 0.))
476  {
477  gains << gain_x_k/gain_x_b, 1., gain_x_m/gain_x_b, 0., 0., 0.,
478  0., 0., 0., gain_y_k, 0., gain_y_m;
479  selectxy_mat << 0., 1./gain_x_b, 0., 0,
480  0., 0., 0., 1.;
481  }
482  else
483  {
484  gains << gain_x_k/gain_x_b, 1., gain_x_m/gain_x_b, 0., 0., 0.,
485  0., 0., 0., gain_y_k/gain_y_b, 1., gain_y_m/gain_y_b;
486  selectxy_mat << 0., 1./gain_x_b, 0., 0,
487  0., 0., 0., 1./gain_y_b;
488  }
489  Gmbk_ = etools::makeBlockDiagonal(gains, (mpc_parameters_.preview_horizon_length_));
490  }
