240  HUMOTO_ASSERT(sfsm_params_.num_steps_ != 1, "Unsupported number of steps.");
241 
242  ss_states_to_termination_ = sfsm_params_.num_steps_;
243  current_time_ms_ = 0;
244  is_tds_started_ = false;
245  current_stance_.previous_nontds_stance_type_ = StanceType::UNDEFINED;
246  current_stance_.subtype_ = StanceSubType::FIRST;
247  current_stance_.type_ = sfsm_params_.first_stance_;
248  current_stance_.duration_ms_ = sfsm_params_.first_stance_duration_ms_;
249 
250  switch (current_stance_.type_)
251  {
252  case StanceType::LSS:
253  case StanceType::RSS:
254  current_stance_.previous_nonds_stance_type_ = current_stance_.type_;
255  break;
256  case StanceType::DS:
257  switch(sfsm_params_.first_ss_type_)
258  {
259  case StanceType::RSS:
260  current_stance_.previous_nonds_stance_type_ = StanceType::LSS;
261  break;
262  case StanceType::LSS:
263  current_stance_.previous_nonds_stance_type_ = StanceType::RSS;
264  break;
265  case StanceType::UNDEFINED:
266  current_stance_.previous_nonds_stance_type_ = StanceType::UNDEFINED;
267  sfsm_params_.num_steps_ = 0;
268  break;
269  default:
270  HUMOTO_THROW_MSG("Misformed walking options.");
271  break;
272  }
273  break;
274  default:
275  HUMOTO_THROW_MSG("Misformed walking options.");

312  /**
313  * @brief Preview sequence of Stances of the FSM
314  *
315  * @param[in] preview_duration_ms
316  *
317  * @return
318  */
319  std::vector<Stance> StanceFiniteStateMachine::previewStances(const std::size_t preview_duration_ms) const
320  {
321  bool not_enough_states = false;
322 
323  std::vector<Stance> stance_types;
324  Stance stance_type;
325 
326  StanceFiniteStateMachine stance_fsm = StanceFiniteStateMachine(*this);
327 
328 
329  for (std::size_t duration_ms = preview_duration_ms; duration_ms > 0; )
330  {
331  stance_type.type_ = stance_fsm.current_stance_.type_;
332  stance_type.subtype_ = stance_fsm.current_stance_.subtype_;
333  stance_type.duration_ms_ = std::min(stance_fsm.current_stance_.duration_ms_ - stance_fsm.current_time_ms_, duration_ms);
334  stance_type.total_duration_ms_ = stance_fsm.current_stance_.duration_ms_;
335  stance_type.previous_nonds_stance_type_ = stance_fsm.current_stance_.previous_nonds_stance_type_;
336  stance_type.previous_nontds_stance_type_ = stance_fsm.current_stance_.previous_nontds_stance_type_;
337 
338  if(stance_fsm.current_time_ms_ + duration_ms <= stance_fsm.current_stance_.duration_ms_)
339  {
340  duration_ms = 0;
341  stance_fsm.current_time_ms_ = stance_fsm.current_time_ms_ + duration_ms;
342  }
343  else
344  {
345  duration_ms -= (stance_fsm.current_stance_.duration_ms_ - stance_fsm.current_time_ms_);
346 
347  if((stance_fsm.ss_states_to_termination_ == 0) && (stance_fsm.current_stance_.type_ == StanceType::DS))
348  {
349  not_enough_states = true;
350  break;
351  }
352  else
353  {
354  stance_fsm.shiftStance();
355  }
356  }
357 
358  stance_types.push_back(stance_type);
359  }
360 
361  //always add one state afted TDS
362  if((not_enough_states == false) && (stance_types.back().type_ == StanceType::TDS))
363  {
364  if(stance_fsm.current_stance_.type_ == StanceType::TDS)
365  {
366  stance_fsm.shiftStance();
367  }
368 
369  stance_type.type_ = stance_fsm.current_stance_.type_;
370  stance_type.subtype_ = stance_fsm.current_stance_.subtype_;
371  stance_type.duration_ms_ = 0;
372  stance_type.total_duration_ms_ = 0;
373  stance_type.previous_nonds_stance_type_ = stance_fsm.current_stance_.previous_nonds_stance_type_;
374  stance_type.previous_nontds_stance_type_ = stance_fsm.current_stance_.previous_nontds_stance_type_;
375 
376  stance_types.push_back(stance_type);
377  }
378 
379  return stance_types;
380  }

431  /**
432  * @brief Shift state of the model
433  */
434  void StanceFiniteStateMachine::shiftStance()
435  {
436  is_tds_started_ = true;
437 
438  current_stance_.subtype_ = StanceSubType::INTERMEDIATE;
439 
440  switch(current_stance_.type_)
441  {
442  case StanceType::LSS:
443  current_stance_.previous_nonds_stance_type_ = current_stance_.type_;
444  current_stance_.previous_nontds_stance_type_ = current_stance_.type_;
445  current_stance_.type_ = StanceType::TDS;
446  current_stance_.duration_ms_ = sfsm_params_.getDurationMs(StanceType::TDS);
447  current_time_ms_ = 0;
448  break;
449 
450  case StanceType::RSS:
451  current_stance_.previous_nonds_stance_type_ = current_stance_.type_;
452  current_stance_.previous_nontds_stance_type_ = current_stance_.type_;
453  current_stance_.type_ = StanceType::TDS;
454  current_stance_.duration_ms_ = sfsm_params_.getDurationMs(StanceType::TDS);
455  current_time_ms_ = 0;
456  break;
457 
458  case StanceType::TDS:
459  is_tds_started_ = false;
460 
461  if(ss_states_to_termination_ == 0)
462  {
463  current_stance_.subtype_ = StanceSubType::LAST;
464  current_stance_.type_ = StanceType::DS;
465  current_stance_.duration_ms_ = sfsm_params_.getDurationMs(StanceType::DS);
466  current_time_ms_ = 0;
467  }
468  else
469  {
470  if(ss_states_to_termination_ > 0)
471  {
472  ss_states_to_termination_ = ss_states_to_termination_ - 1;
473  }
474 
475  if(current_stance_.previous_nonds_stance_type_ == StanceType::LSS)
476  {
477  current_stance_.type_ = StanceType::RSS;
478  current_stance_.duration_ms_ = sfsm_params_.getDurationMs(StanceType::RSS);
479  current_time_ms_ = 0;
480  }
481  else if(current_stance_.previous_nonds_stance_type_ == StanceType::RSS)
482  {
483  current_stance_.type_ = StanceType::LSS;
484  current_stance_.duration_ms_ = sfsm_params_.getDurationMs(StanceType::LSS);
485  current_time_ms_ = 0;
486  }
487  else
488  {
489  HUMOTO_THROW_MSG("FSM is in an incorrect state.");
490  }
491  }
492  break;
493 
494  case StanceType::DS:
495  current_stance_.previous_nontds_stance_type_ = current_stance_.type_;
496 
497  if(ss_states_to_termination_ == 0)
498  {
499  current_stance_.type_ = StanceType::DS;
500  current_stance_.duration_ms_ = sfsm_params_.getDurationMs(StanceType::DS);
501  current_time_ms_ = 0;
502  }
503  else
504  {
505  current_stance_.type_ = StanceType::TDS;
506  current_stance_.duration_ms_ = sfsm_params_.getDurationMs(StanceType::TDS);
507  current_time_ms_ = 0;
508  }
509  break;
510 
511  default:
512  HUMOTO_THROW_MSG("FSM is in an incorrect state.");
513  }
514  }
