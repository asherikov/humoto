33  /**
34  * @brief Reorder the wrench data for ease of use in the MPC
35  *
36  * @param[in] full_predicted
37  *
38  * @return n_stepx4 vector with the order: torque_y, force_x, torquex, force_y
39  */
40  Eigen::VectorXd getReorderedSet(const Eigen::VectorXd &full_predicted)
41  {
42  Eigen::VectorXd reordered;
43  unsigned int n_steps = full_predicted.rows()/6;
44  reordered.resize(4*n_steps);
45  for(unsigned int i=0; i<n_steps; ++i)
46  {
47  reordered(i*4+0) = full_predicted(i*6+4); //torque_y
48  reordered(i*4+1) = full_predicted(i*6+0); //force_x
49  reordered(i*4+2) = full_predicted(i*6+3); //torque_x
50  reordered(i*4+3) = full_predicted(i*6+1); //force_y
51  }
52  return reordered;
