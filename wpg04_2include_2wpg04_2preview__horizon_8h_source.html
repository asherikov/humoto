225  /**
226  * @brief Preview a sequence of walk states with velocities, rotations etc.
227  *
228  * @param[in] model
229  * @param[in] stance_fsm
230  * @param[in] walk_parameters
231  * @param[in] preview_duration
232  */
233  void PreviewHorizon::previewStates(const Model &model,
234  const humoto::walking::StanceFiniteStateMachine &stance_fsm,
235  const WalkParameters &walk_parameters,
236  const std::size_t preview_duration)
237  {
238  std::vector<humoto::walking::Stance> stances;
239 
240  stances = stance_fsm.previewStances(preview_duration);
241 
242  double lss_theta = model.getFootState(humoto::LeftOrRight::LEFT).rpy_(humoto::AngleIndex::YAW);
243  double rss_theta = model.getFootState(humoto::LeftOrRight::RIGHT).rpy_(humoto::AngleIndex::YAW);
244  double ds_theta = (lss_theta + rss_theta) / 2.;
245 
246  walk_states_.resize(stances.size());
247  for(std::size_t i = 0; i < stances.size(); ++i)
248  {
249  WalkState state;
250  state = stances.at(i);
251 
252  switch(state.type_)
253  {
254  case humoto::walking::StanceType::LSS:
255  state.theta_ = lss_theta;
256  state.cvel_ref_ = walk_parameters.com_velocity_;
257  break;
258 
259  case humoto::walking::StanceType::RSS:
260  state.theta_ = rss_theta;
261  state.cvel_ref_ = walk_parameters.com_velocity_;
262  break;
263 
264  case humoto::walking::StanceType::DS:
265  state.theta_ = ds_theta;
266  if (state.subtype_ == humoto::walking::StanceSubType::FIRST)
267  {
268  state.cvel_ref_ = walk_parameters.first_stance_com_velocity_;
269  }
270  else if(state.subtype_ == humoto::walking::StanceSubType::LAST)
271  {
272  state.cvel_ref_ = walk_parameters.last_stance_com_velocity_;
273  }
274  else
275  {
276  HUMOTO_THROW_MSG("Unsupported stance subtype.");
277  }
278  break;
279 
280  case humoto::walking::StanceType::TDS:
281  state.cvel_ref_ = walk_parameters.com_velocity_;
282  state.theta_ = ds_theta;
283  if((stances.at(i + 1).type_ == humoto::walking::StanceType::RSS) &&
284  (state.previous_nontds_stance_type_ != humoto::walking::StanceType::DS))
285  {
286  rss_theta = lss_theta + walk_parameters.theta_increment_;
287  state.theta_ = rss_theta;
288  }
289  if((stances.at(i + 1).type_ == humoto::walking::StanceType::LSS) &&
290  (state.previous_nontds_stance_type_ != humoto::walking::StanceType::DS))
291  {
292  lss_theta = rss_theta + walk_parameters.theta_increment_;
293  state.theta_ = lss_theta;
294  }
295  if((stances.at(i + 1).type_ == humoto::walking::StanceType::DS) &&
296  (state.previous_nontds_stance_type_ != humoto::walking::StanceType::LSS))
297  {
298  state.cvel_ref_ = walk_parameters.last_stance_com_velocity_;
299  state.theta_ = rss_theta;
300  ds_theta = rss_theta;
301  }
302  if((stances.at(i + 1).type_ == humoto::walking::StanceType::DS) &&
303  (state.previous_nontds_stance_type_ != humoto::walking::StanceType::RSS))
304  {
305  state.cvel_ref_ = walk_parameters.last_stance_com_velocity_;
306  state.theta_ = lss_theta;
307  ds_theta = lss_theta;
308  }
309  break;
310 
311  default :
312  HUMOTO_THROW_MSG("Unsupported stance type.");
313  break;
314  }
315  state.rotation_ = Eigen::Rotation2Dd(state.theta_);
316  state.R_cvel_ref_.noalias() = state.rotation_ * state.cvel_ref_;
317  state.step_height_ = 0.;
318 
319  if (humoto::walking::StanceType::TDS != state.type_)
320  {
321  getConstraints(state, model);
322  }
323  walk_states_[i] = state;
324  }
325  }

341  /**
342  * @brief Form the preview horizon object
343  *
344  * @param[in] mpc_params
345  * @param[in] model
346  * @param[in] stance_fsm
347  * @param[in] walk_parameters
348  *
349  * @return
350  */
351  bool PreviewHorizon::form(const MPCParameters &mpc_params,
352  const Model &model,
353  const humoto::walking::StanceFiniteStateMachine &stance_fsm,
354  const WalkParameters &walk_parameters)
355  {
356  bool preview_horizon_formed = true;
357 
358  // Length of the preview horizon (N)
359  std::size_t preview_horizon_length_ = mpc_params.preview_horizon_length_;
360  std::size_t sampling_time_ms_ = mpc_params.sampling_time_ms_;
361  std::size_t walk_index = 0;
362  std::size_t num_intervals_left = 0;
363 
364  //is_new_sample_ = stance_fsm.is_new_sample_;
365  previewStates(model, stance_fsm, walk_parameters, preview_horizon_length_*sampling_time_ms_);
366 
367  if(walk_states_.empty())
368  {
369  preview_horizon_formed = false;
370  return (preview_horizon_formed);
371  }
372 
373  variable_steps_indices_.clear();
374 
375  std::vector<std::size_t> support_indices;
376 
377  intervals_.resize(preview_horizon_length_);
378 
379  std::size_t interval_index = 0;
380  while(interval_index < preview_horizon_length_)
381  {
382  WalkState state = walk_states_.at(walk_index);
383 
384  PreviewHorizonInterval interval;
385  interval.cvel_ref_ = state.R_cvel_ref_;
386  interval.com_height_ = model.getCoMHeight();
387  interval.omega_ = model.getOmega(interval.com_height_);
388 
389  support_indices.push_back(interval_index);
390 
391  if (walk_index == 0)
392  {
393  // current sampling interval (the 1st in the preview window)
394  if (state.type_ == humoto::walking::StanceType::TDS)
395  {
396  interval.T_ms_ = state.duration_ms_;
397  interval.T_ = convertMillisecondToSecond(interval.T_ms_);
398 
399  ++walk_index;
400  state = walk_states_.at(walk_index);
401  num_intervals_left = floor(state.duration_ms_ / mpc_params.sampling_time_ms_);
402  }
403  else
404  {
405  interval.T_ms_ = state.duration_ms_ % mpc_params.sampling_time_ms_;
406  interval.T_ = convertMillisecondToSecond(interval.T_ms_);
407  num_intervals_left = floor(state.duration_ms_ / mpc_params.sampling_time_ms_);
408 
409  if ((state.duration_ms_ % mpc_params.sampling_time_ms_) == 0)
410  {
411  interval.T_ms_ = mpc_params.sampling_time_ms_;
412  interval.T_ = convertMillisecondToSecond(interval.T_ms_);
413 
414  --num_intervals_left;
415  }
416  }
417 
418  interval.state_index_ = walk_index;
419  intervals_[interval_index] = interval;
420  ++interval_index;
421  }
422  else
423  {
424  HUMOTO_ASSERT(humoto::walking::StanceType::TDS == state.type_, "Unexpected state type!");
425 
426  // new variable footstep
427  variable_steps_indices_.push_back(interval_index);
428 
429  // transitional DS
430  interval.T_ms_ = mpc_params.tds_sampling_time_ms_;
431  interval.T_ = convertMillisecondToSecond(interval.T_ms_);
432 
433  // next SS
434  ++walk_index;
435  state = walk_states_.at(walk_index);
436 
437  interval.state_index_ = walk_index;
438  intervals_[interval_index] = interval;
439  ++interval_index;
440 
441  num_intervals_left = floor(state.duration_ms_ / mpc_params.sampling_time_ms_);
442  }
443 
444  interval.T_ms_ = mpc_params.sampling_time_ms_;
445  interval.T_ = convertMillisecondToSecond(interval.T_ms_);
446 
447  // fill the other intervals
448  std::size_t max_index = std::min(preview_horizon_length_, interval_index + num_intervals_left);
449  for(std::size_t i = interval_index; i < max_index; ++i)
450  {
451  intervals_[i] = interval;
452  }
453 
454  interval_index += num_intervals_left;
455  ++walk_index;
456  }
457 
458  return (preview_horizon_formed);
459  }

468  void PreviewHorizon::getConstraints(WalkState &state,

470  {
471  if(state.subtype_ == humoto::walking::StanceSubType::FIRST)
472  {
473  state.fd_bounds_.setZero();
474  }
475  else
476  {
477  switch(state.type_)
478  {
479  case humoto::walking::StanceType::LSS:
480  switch(state.previous_nontds_stance_type_)
481  {
482  case humoto::walking::StanceType::DS:
483  state.fd_bounds_ = model.getFootBounds(humoto::LeftOrRight::LEFT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_ADS);
484  break;
485  case humoto::walking::StanceType::RSS:
486  state.fd_bounds_ = model.getFootBounds(humoto::LeftOrRight::LEFT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_SS);
487  break;
488  default:
489  HUMOTO_THROW_MSG("unexpected support type");
490  break;
491  }
492  break;
493 
494  case humoto::walking::StanceType::RSS:
495  switch(state.previous_nontds_stance_type_)
496  {
497  case humoto::walking::StanceType::DS:
498  state.fd_bounds_ = model.getFootBounds(humoto::LeftOrRight::RIGHT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_ADS);
499  break;
500  case humoto::walking::StanceType::LSS:
501  state.fd_bounds_ = model.getFootBounds(humoto::LeftOrRight::RIGHT, humoto::walking::FootBoundsType::WITH_RESPECT_TO_SS);
502  break;
503  default:
504  HUMOTO_THROW_MSG("unexpected support type");
505  break;
506  }
507  break;
508 
509  case humoto::walking::StanceType::DS:
510  switch(state.previous_nontds_stance_type_)
511  {
512  case humoto::walking::StanceType::LSS:
513  state.fd_bounds_ = model.getFootBounds(humoto::LeftOrRight::RIGHT, humoto::walking::FootBoundsType::ALIGNED_TO_SS);
514  break;
515  case humoto::walking::StanceType::RSS:

517  break;
518  default:
519  HUMOTO_THROW_MSG("unexpected support type");
520  break;
521  }

526  break;

528  }

530  if (humoto::walking::StanceType::DS == state.type_)
531  {
532  state.cop_bounds_ = model.getADSCoPBounds();
