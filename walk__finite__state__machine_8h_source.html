227  * @brief Initialize the FSM
228  *
229  * @param[in] model
230  * @param[in] mpc_param
231  * @param[in] walk_options
232  */
233  void WalkFiniteStateMachine::initialize(const Model &model,
234  const MPCParameters &mpc_param,
235  const WalkOptions &walk_options)
236  {
237  if((walk_options.ss_duration_ms_ % mpc_param.sampling_time_ms_) != 0 ||
238  (walk_options.tds_duration_ms_ != mpc_param.tds_sampling_time_ms_) ||
239  (walk_options.first_ds_duration_ms_ % mpc_param.sampling_time_ms_) != 0 ||
240  (walk_options.last_ds_duration_ms_ % mpc_param.sampling_time_ms_) != 0)
241  {
242  HUMOTO_THROW_MSG("Incorrectly specified duration of a state!");
243  }
244 
245  if (walk_options.num_steps_ == 1)
246  {
247  HUMOTO_THROW_MSG("Unsupported number of steps.");
248  }
249 
250  states_.clear();
251 
252  /// Assume aligned feet in DS
253  init_theta_ = ( model.left_foot_.rpy_(AngleIndex::YAW) + model.right_foot_.rpy_(AngleIndex::YAW) ) / 2.;
254 
255  WalkState state;
256 
257 
258  state.type_ = model.state_type_;
259 
260  if (state.type_ != STATE_DS)
261  {
262  HUMOTO_THROW_MSG("Initial state must be DS!");
263  }
264 
265  state.duration_ms_ = walk_options.first_ds_duration_ms_;
266  state.cvel_ref_ = walk_options.first_ds_com_velocity_;
267  state.theta_ = init_theta_;
268 
269  state.rotation_ = Eigen::Rotation2Dd(state.theta_);
270  state.R_cvel_ref_ = state.rotation_ * state.cvel_ref_;
271 
272  state.step_height_ = 0.;
273  states_.push_back(state);
274 
275  if (walk_options.num_steps_ > 0)
276  {
277  state.type_ = STATE_TDS;
278  state.duration_ms_ = walk_options.tds_duration_ms_;
279  state.cvel_ref_ = walk_options.com_velocity_;
280  state.theta_ = init_theta_;
281 
282  state.rotation_ = Eigen::Rotation2Dd(state.theta_);
283  state.R_cvel_ref_ = state.rotation_ * state.cvel_ref_;
284 
285  state.step_height_ = 0.;
286 
287  states_.push_back(state);
288 
289 
290  state.type_ = walk_options.first_ss_type_;
291  state.duration_ms_ = walk_options.ss_duration_ms_;
292  state.step_height_ = walk_options.step_height_;
293 
294  states_.push_back(state);
295  }
296 
297  for(unsigned int i=2; i<walk_options.num_steps_ - 2; ++i)
298  {
299  ModelStateType next_ss_type;
300  if(states_.back().type_ == STATE_RSS)
301  {
302  next_ss_type = STATE_LSS;
303  }
304  else
305  {
306  next_ss_type = STATE_RSS;
307  }
308 
309  state.type_ = STATE_TDS;
310  state.duration_ms_ = walk_options.tds_duration_ms_;
311  state.cvel_ref_ = walk_options.com_velocity_;
312  state.theta_ = states_.back().theta_ + walk_options.theta_inc_;
313 
314  state.rotation_ = Eigen::Rotation2Dd(state.theta_);
315  state.R_cvel_ref_ = state.rotation_ * state.cvel_ref_;
316 
317  state.step_height_ = 0.;
318 
319  states_.push_back(state);
320 
321 
322  state.type_ = next_ss_type;
323  state.duration_ms_ = walk_options.ss_duration_ms_;
324  state.step_height_ = walk_options.step_height_;
325 
326  states_.push_back(state);
327  }
328 
329  if (walk_options.num_steps_ > 0)
330  {
331  state.type_ = STATE_TDS;
332  state.duration_ms_ = walk_options.tds_duration_ms_;
333  state.cvel_ref_ = walk_options.last_ds_com_velocity_;
334  state.theta_ = states_.back().theta_;
335 
336  state.rotation_ = Eigen::Rotation2Dd(state.theta_);
337  state.R_cvel_ref_ = state.rotation_ * state.cvel_ref_;
338 
339  state.step_height_ = 0.;
340 
341  states_.push_back(state);
342 
343 
344  state.type_ = STATE_DS;
345  state.duration_ms_ = walk_options.last_ds_duration_ms_;
346 
347  states_.push_back(state);
348  }
349 
350 
351  index_ = 0; // current step index
352 
353  is_new_sample_ = true;
354  is_end_of_ss_ = false;
355  is_end_of_fixed_ds_ = false;
356  is_last_ss = false;
357 
358  current_step_duration_ms_ = states_.at(index_).duration_ms_;
359  }

363  /**
364  * @brief Set a new target com velocity
365  */
366  void WalkFiniteStateMachine::setCoMVelocity(const etools::Vector2 &com_velocity)
367  {
368  for(unsigned int i=index_; i < states_.size(); ++i)
369  {
370  switch(states_.at(i).type_)
371  {
372  case STATE_RSS:
373  case STATE_LSS:
374  states_.at(i).cvel_ref_ = com_velocity;
375  states_.at(i).R_cvel_ref_ = states_.at(i).rotation_ * com_velocity;
376  break;
377  case STATE_TDS:
378  /// this also changes the com_velocity of the last TDS (it should not)
379  states_.at(i).cvel_ref_ = com_velocity;
380  states_.at(i).R_cvel_ref_ = states_.at(i).rotation_ * com_velocity;
381  break;
382  default:
383  // do nothing
384  break;
385  }
386  }
387  }

391  /**
392  * @brief Set a new theta increment for turning the robot
393  */
394  void WalkFiniteStateMachine::setThetaIncrement(const double &theta_inc)
395  {
396  // initialize from the current state
397  double current_theta = states_.at(index_).theta_;
398  etools::Matrix2 current_rotation = states_.at(index_).rotation_;
399 
400  // update future states
401  for(unsigned int i=index_; i < states_.size(); ++i)
402  {
403  switch(states_.at(i).type_)
404  {
405  case STATE_RSS:
406  case STATE_LSS:
407  states_.at(i).theta_ = current_theta;
408  states_.at(i).rotation_ = current_rotation;
409  states_.at(i).R_cvel_ref_ = current_rotation * states_.at(i).cvel_ref_;
410  break;
411  case STATE_TDS:
412  current_theta += theta_inc;
413  current_rotation = Eigen::Rotation2Dd(current_theta);
414 
415  states_.at(i).theta_ = current_theta;
416  states_.at(i).rotation_ = current_rotation;
417  states_.at(i).R_cvel_ref_ = current_rotation * states_.at(i).cvel_ref_;
418  break;
419  default:
420  // do nothing
421  break;
422  }
423  }

428  /**
429  * @brief Stops the walking PG by removing states that are safe to remove
430  */
431  void WalkFiniteStateMachine::stopWalking()
432  {
433  // Make sure there are states to be removed
434  if(index_ < (states_.size()-7))
435  {
436  ModelStateType last_ss_type = states_.at(states_.size() - 3).type_;
437  ModelStateType current_state_type = states_.at(index_).type_;
438  ModelStateType next_state_type = states_.at(index_+1).type_;
439  unsigned int states_to_keep = 0;
440 
441  switch(current_state_type)
442  {
443  case STATE_RSS:
444  case STATE_LSS:
445  if(current_state_type == last_ss_type)
446  {
447  states_to_keep = 6;
448  }
449  else
450  {
451  states_to_keep = 4;
452  }
453  break;
454  case STATE_TDS:
455  if(next_state_type == last_ss_type)
456  {
457  states_to_keep = 3;
458  }
459  else
460  {
461  states_to_keep = 5;
462  }
463  break;
464  default:
465  // do nothing
466  break;
467  }
468  // fast-forward the current state to a similar one near the end
469  states_.erase(states_.begin()+index_+1, states_.end()-states_to_keep);
470 
471  // Correct rotation of last states
472  /// velocities might need correcting
473  double current_theta = states_.at(index_).theta_;
474  etools::Matrix2 current_rotation = states_.at(index_).rotation_;
475  for(unsigned int i=states_.size()-states_to_keep; i<states_.size(); ++i)
476  {
477  states_.at(i).theta_ = current_theta;
478  states_.at(i).rotation_ = current_rotation;
479  }
480  }
481  }

485  /**
486  * @brief Update the state machine
487  */
488  void WalkFiniteStateMachine::update(const MPCParameters &mpc_param)
489  {
490  is_end_of_ss_ = false;
491  is_end_of_fixed_ds_ = false;
492  is_new_sample_ = false;
493 
494 
495  current_step_duration_ms_ = current_step_duration_ms_ - mpc_param.subsampling_time_ms_;
496 
497  if ((current_step_duration_ms_ % mpc_param.sampling_time_ms_) == 0)
498  {
499  is_new_sample_ = true;
500  }
501 
502  if (current_step_duration_ms_ == 0)
503  {
504  index_ = index_ + 1;
505  current_step_duration_ms_ = states_.at(index_).duration_ms_;
506 
507 
508  switch(states_.at(index_-1).type_)
509  {
510  case STATE_RSS:
511  case STATE_LSS:
512  is_end_of_ss_ = true;
513  break;
514  case STATE_DS:
515  is_end_of_fixed_ds_ = true;
516  break;
517  default:
518  // do nothing
519  break;
520  }
521  }
522 
523  // This assumes that the ending state sequence is: SS - TDS - DS
524  if (index_ == states_.size() - 3)
525  {
526  is_last_ss = true;
527  }
528  }
